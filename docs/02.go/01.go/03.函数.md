## 自定义函数

**函数名首字母小写为private,大写即为public**

### 无参无返回值

```
package main

import "fmt"

func main() {
	test()
}

func test() {
	fmt.Println("hello")
}
输出:
hello
```

### 有参无返回值

#### 普通参数列表

```
package main

import "fmt"

func main() {
	test(666, "shirs")
}

func test(a int, b string) {
	fmt.Printf("a=%v,b=%v\n", a, b)
}
//同类型可以这样定义，ab都是string类型
func test2(a, b string) {
	fmt.Printf("a=%v,b=%v\n", a, b)
}
输出:
a=666,b=shirs
```

#### 不定参数类型

```
package main

import "fmt"

func main() {
	//不定参数可以不传参
	test()
	fmt.Println("---------------")
	test(5, 6, 7)
	fmt.Println("---------------")
	test_v2("xian", 6)
}

//不定参数类型,   ...type
func test(args ...int) {
	for i, data := range args {
		fmt.Printf("i=%v,data=%v\n", i, data)
	}
}

//不定参数只能放在形参的最后一个
func test_v2(str string, args ...int) {
	for i, data := range args {
		fmt.Printf("str=%v,i=%v,data=%v\n", str, i, data)
	}
}
输出：
---------------
i=0,data=5
i=1,data=6
i=2,data=7
---------------
str=xian,i=0,data=6

```

#### 不定参数传递

```

### 无参有返回值


#### 一个返回值

```
package main

import "fmt"

func main() {
	result := test()
	fmt.Printf("result=%v\n", result)
	fmt.Println("-----------------")
	result = test_v2()
	fmt.Printf("result=%v\n", result)
}

func test() int {
	return 666
}

//go推荐写法，给返回值起一个变量名
func test_v2() (result int) {
	result = 777
	return
}
输出:
result=666
-----------------
result=777
```


#### 多个返回值


```
package main

import "fmt"

func main() {
	a, b, c := test()
	fmt.Printf("a=%v,b=%v,c=%v\n", a, b, c)
}

func test() (a, b, c int) {
	a, b, c = 7, 8, 9
	return
}
输出:

a=7,b=8,c=9
```


## 递归函数

```
package main

import "fmt"

func main() {
	sum := test(1)
	fmt.Printf("a=%v\n", sum)
}

//计算1+2+3+...+100
func test(a int) (sum int) {
	if a == 100 {
		return 100
	}
	sum = a + test(a+1)

	return
}
输出：
5050
```
## 回调函数和函数类型(重要)

```
package main

import "fmt"

func main() {
	result := Calculate(29, 11, add)
	fmt.Printf("add:result=%v\n", result)
	result = Calculate(29, 11, minus)
	fmt.Printf("minus:result=%v\n", result)
}

func add(a, b int) (res int) {
	res = a + b
	return
}

func minus(a, b int) (res int) {
	res = a - b
	return
}

//函数也是一种数据类型，可以通过type给一个函数类型起名
//通过参数个数和类型和返回值确定一个函数？
type FuncType func(a, b int) int //没有函数名字，没有函数体

//回调函数，函数有一个参数是函数类型
//多态，接口的不同实现
//可以先指定函数，不做具体实现
func Calculate(a, b int, calc FuncType) (result int) {
	result = calc(a, b)
	return
}
输出:
add:result=40
minus:result=18

```

## 匿名函数和闭包

<img src="/old-times/png/go/go_bibao.png">

```
package main

import "fmt"

func main() {
	a := 10
	name := "shirs"

	//匿名函数，没有函数名字，定义必须被使用(定义/调用),否则编译不通过
	f1 := func() {
		fmt.Printf("一、a=%v,name=%v\n", a, name)
	}
	//调用
	f1()
	fmt.Println("-----给匿名函数起别名----")
	//给匿名函数起别名
	type FuncType func()
	/*var f2 FuncType
	f2=f1
	f2()*/
	f2 := f1
	f2()

	fmt.Println("-----定义匿名函数同时调用----")
	//定义匿名函数同时调用
	func() {
		fmt.Printf("二、a=%v,name=%v\n", a, name)
	}() //()即代表了调用

	fmt.Println("-----带参数的匿名函数----")
	//带参数的匿名函数
	func(i, j int) {
		fmt.Printf("三、i=%v,j=%v\n", i, j)
	}(10, 100)

	fmt.Println("-----有参有返回值的匿名函数----")
	//有参有返回值的匿名函数
	max, min := func(i, j int) (max, min int) {
		if i > j {
			max = i
			min = j
		} else {
			max = j
			min = i
		}
		return 
	}(10, 100)

	fmt.Printf("四、max=%v,min=%v\n", max, min)
}
输出:

一、a=10,name=shirs
-----给匿名函数起别名----
一、a=10,name=shirs
-----定义匿名函数同时调用----
二、a=10,name=shirs
-----带参数的匿名函数----
三、i=10,j=100
-----有参有返回值的匿名函数----
四、max=100,min=10
```


### 闭包捕获外部变量

```
package main

import "fmt"

func main() {
	a := 10
	name := "shirs"
	fmt.Printf("原来的:a=%v,name=%v\n", a, name)
	//闭包以引用方式捕获外部变量
	func() {
		a = 100
		name = "xian"
		fmt.Printf("匿名函数内:a=%v,name=%v\n", a, name)
	}()//()代表直接调用
	fmt.Printf("匿名函数修改后:a=%v,name=%v\n", a, name)
}

```

## 关键字defer->延迟调用

### defer基础使用
*延迟调用，程序结束前(正常运行结束/异常报错结束，报错后面的defer不再执行)调用*

```
package main

import "fmt"

func main() {
	//延迟调用，main方法结束前调用
	defer fmt.Println("bbbbbbbbbbbbb")
	fmt.Println("aaaaaaaaaaaaa")
}

输出:
aaaaaaaaaaaaa
bbbbbbbbbbbbb
```

### 多个defer调用顺序

如果一个函数中又多个defer语句，它们会以后进先出的顺序执行。    
某个函数或延迟调用发生错误，**错误前**的defer仍旧会被执行

### defer和匿名函数结合使用

```
package main

import "fmt"

func main() {
	a := 10
	b := 20
	defer func() {
		//毫无疑问.为变量更改后的值
		fmt.Printf("inner:a=%d,b=%d\n", a, b)
	}()
	a = 111
	b = 222
	fmt.Printf("outer:a=%d,b=%d\n", a, b)
}

输出:
outer:a=111,b=222
inner:a=111,b=222
